062@Slf4j
063@AllArgsConstructor
064public class DefaultResponseLocatorCapturingHandler extends SimpleChannelInboundHandler<FullHttpRequest> {
065
066    private final HttpApiHandler api;
067
068    @Override
069    protected void channelRead0(final ChannelHandlerContext ctx, final FullHttpRequest request) {
070        if (HttpMethod.CONNECT.name().equalsIgnoreCase(request.method().name())) {
071            final FullHttpResponse response =
072                    new DefaultFullHttpResponse(HttpVersion.HTTP_1_1, HttpResponseStatus.OK, Unpooled.EMPTY_BUFFER);
073            if (api.getSslContext() != null) {
074                final SSLEngine sslEngine = api.getSslContext().createSSLEngine();
075                sslEngine.setUseClientMode(false);
076                ctx.channel().pipeline().addFirst("ssl", new SslHandler(sslEngine, true));
077
078                final String uri = request.uri();
079                final String[] parts = uri.split(":");
080                ctx.channel().attr(BASE).set(
081                        "https://" + parts[0] + (parts.length > 1 && !"443".equals(parts[1]) ? ":" + parts[1] : ""));
082            }
083            ctx.writeAndFlush(response);
084            return;
085        }
086
087        api.getExecutor().execute(() -> {
088            final Attribute<String> baseAttr = ctx.channel().attr(Handlers.BASE);
089
090            final DefaultResponseLocator.RequestModel requestModel = new DefaultResponseLocator.RequestModel();
091            requestModel.setMethod(request.method().name());
092            requestModel.setUri((baseAttr == null || baseAttr.get() == null ? "" : baseAttr.get()) + request.uri());
093            requestModel.setHeaders(StreamSupport
094                    .stream(Spliterators.spliteratorUnknownSize(request.headers().iteratorAsString(),
095                            Spliterator.IMMUTABLE), false)
096                    .filter(h -> !api.getHeaderFilter().test(h.getKey()))
097                    .collect(toMap(Map.Entry::getKey, Map.Entry::getValue)));
098            final DefaultResponseLocator.Model model = new DefaultResponseLocator.Model();
099            model.setRequest(requestModel);
100
101            // do the remote request with all the incoming data and save it
102            // note: this request must be synchronous for now
103            final Response resp;
104            try {
105                final URL url = new URL(requestModel.getUri());
106                final HttpURLConnection connection = HttpURLConnection.class.cast(url.openConnection(Proxy.NO_PROXY));
107                connection.setConnectTimeout(30000);
108                connection.setReadTimeout(20000);
109                if (HttpsURLConnection.class.isInstance(connection)) {
110                    final HttpsURLConnection httpsURLConnection = HttpsURLConnection.class.cast(connection);
111                    httpsURLConnection.setHostnameVerifier((h, s) -> true);
112                    httpsURLConnection.setSSLSocketFactory(api.getSslContext().getSocketFactory());
113                }
114                if (request.method() != null) {
115                    final String requestMethod = request.method().name();
116                    connection.setRequestMethod(requestMethod);
117
118                    if (!"HEAD".equalsIgnoreCase(requestMethod) && request.content().readableBytes() > 0) {
119                        connection.setDoOutput(true);
120                        request.content().readBytes(connection.getOutputStream(), request.content().readableBytes());
121                    }
122                }
123
124                final int responseCode = connection.getResponseCode();
125                final int defaultLength =
126                        ofNullable(connection.getHeaderField("content-length")).map(Integer::parseInt).orElse(8192);
127                resp = new ResponseImpl(
128                        connection
129                                .getHeaderFields()
130                                .entrySet()
131                                .stream()
132                                .filter(e -> e.getKey() != null)
133                                .filter(h -> !api.getHeaderFilter().test(h.getKey()))
134                                .collect(toMap(Map.Entry::getKey,
135                                        e -> e.getValue().stream().collect(Collectors.joining(",")))),
136
137                        responseCode, responseCode <= 399 ? slurp(connection.getInputStream(), defaultLength)
138                                : slurp(connection.getErrorStream(), defaultLength));
139            } catch (final Exception e) {
140                log.error(e.getMessage(), e);
141                sendError(ctx, HttpResponseStatus.BAD_REQUEST);
142                return;
143            }
144
145            final DefaultResponseLocator.ResponseModel responseModel = new DefaultResponseLocator.ResponseModel();
146            responseModel.setStatus(resp.status());
147            responseModel.setHeaders(resp.headers());
148            // todo: support as byte[] for not text responses
149            responseModel.setPayload(new String(resp.payload(), StandardCharsets.UTF_8));
150            model.setResponse(responseModel);
151
152            if (DefaultResponseLocator.class.isInstance(api.getResponseLocator())) {
153                DefaultResponseLocator.class.cast(api.getResponseLocator()).getCapturingBuffer().add(model);
154            }
155
156            final ByteBuf bytes = ofNullable(resp.payload()).map(Unpooled::copiedBuffer).orElse(Unpooled.EMPTY_BUFFER);
157            final HttpResponse response =
158                    new DefaultFullHttpResponse(HttpVersion.HTTP_1_1, HttpResponseStatus.valueOf(resp.status()), bytes);
159            HttpUtil.setContentLength(response, bytes.array().length);
160
161            ofNullable(resp.headers()).ifPresent(h -> h.forEach((k, v) -> response.headers().set(k, v)));
162            ctx.writeAndFlush(response);
163        });
164    }
165
166    private byte[] slurp(final InputStream inputStream, final int defaultLen) throws IOException {
167        final ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream(defaultLen);
168        final byte[] bytes = new byte[defaultLen];
169        int read;
170        while ((read = inputStream.read(bytes)) >= 0) {
171            byteArrayOutputStream.write(bytes, 0, read);
172        }
173        return byteArrayOutputStream.toByteArray();
174    }
175
176    @Override
177    public void exceptionCaught(final ChannelHandlerContext ctx, final Throwable cause) {
178        log.error(cause.getMessage(), cause);
179        closeOnFlush(ctx.channel());
180    }
181
182    public static boolean isActive() {
183        return getBaseCapture() != null;
184    }
185
186    // yes, it could be in the API but to avoid to keep it and pass tests in capture mode
187    // we hide it this way for now
188    public static String getBaseCapture() {
189        return ofNullable(System.getProperty("talend.junit.http.capture")).map(value -> {
190            if ("true".equalsIgnoreCase(value)) { // try to guess
191                final File file = new File("src/test/resources");
192                if (file.isDirectory()) {
193                    return file.getAbsolutePath();
194                }
195            }
196            return value;
197        }).orElse(null);
198    }
