016package org.talend.sdk.component.junit.http.internal.impl;
017
018import static java.util.Optional.ofNullable;
019import static java.util.stream.Collectors.toMap;
020import static org.talend.sdk.component.junit.http.internal.impl.Handlers.closeOnFlush;
021import static org.talend.sdk.component.junit.http.internal.impl.Handlers.sendError;
022
023import java.util.Map;
024import java.util.Optional;
025import java.util.Spliterator;
026import java.util.Spliterators;
027import java.util.stream.StreamSupport;
028
029import org.talend.sdk.component.junit.http.api.HttpApiHandler;
030import org.talend.sdk.component.junit.http.api.Response;
031
032import io.netty.buffer.ByteBuf;
033import io.netty.buffer.Unpooled;
034import io.netty.channel.ChannelHandler;
035import io.netty.channel.ChannelHandlerContext;
036import io.netty.channel.SimpleChannelInboundHandler;
037import io.netty.handler.codec.http.DefaultFullHttpResponse;
038import io.netty.handler.codec.http.FullHttpRequest;
039import io.netty.handler.codec.http.HttpResponse;
040import io.netty.handler.codec.http.HttpResponseStatus;
041import io.netty.handler.codec.http.HttpUtil;
042import io.netty.handler.codec.http.HttpVersion;
043
044import lombok.AllArgsConstructor;
045import lombok.extern.slf4j.Slf4j;
046
047@Slf4j
048@AllArgsConstructor
049@ChannelHandler.Sharable
050public class ServingProxyHandler extends SimpleChannelInboundHandler<FullHttpRequest> {
051
052    private final HttpApiHandler api;
053
054    @Override
055    protected void channelRead0(final ChannelHandlerContext ctx, final FullHttpRequest request) {
056        if (!request.decoderResult().isSuccess()) {
057            sendError(ctx, HttpResponseStatus.BAD_REQUEST);
058            return;
059        }
060
061        api.getExecutor().execute(() -> {
062            final Map<String, String> headers = StreamSupport
063                    .stream(Spliterators.spliteratorUnknownSize(request.headers().iteratorAsString(),
064                            Spliterator.IMMUTABLE), false)
065                    .collect(toMap(Map.Entry::getKey, Map.Entry::getValue));
066            final Optional<Response> matching = api.getResponseLocator().findMatching(
067                    new RequestImpl(request.uri(), request.method().name(), headers), api.getHeaderFilter());
068            if (!matching.isPresent()) {
069                sendError(ctx, HttpResponseStatus.BAD_REQUEST);
070                return;
071            }
072
073            final Response resp = matching.get();
074            final ByteBuf bytes = ofNullable(resp.payload()).map(Unpooled::copiedBuffer).orElse(Unpooled.EMPTY_BUFFER);
075            final HttpResponse response =
076                    new DefaultFullHttpResponse(HttpVersion.HTTP_1_1, HttpResponseStatus.valueOf(resp.status()), bytes);
077            HttpUtil.setContentLength(response, bytes.array().length);
078
079            // todo: config for it
080            response.headers().set("X-Talend-Proxy-JUnit", "true");
081
082            ofNullable(resp.headers()).ifPresent(h -> h.forEach((k, v) -> response.headers().set(k, v)));
083            ctx.writeAndFlush(response);
084        });
085    }
086
087    @Override
088    public void exceptionCaught(final ChannelHandlerContext ctx, final Throwable cause) {
089        log.error(cause.getMessage(), cause);
090        closeOnFlush(ctx.channel());
091    }
092}
