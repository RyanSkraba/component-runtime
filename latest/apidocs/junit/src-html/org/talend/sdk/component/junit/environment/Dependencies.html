016package org.talend.sdk.component.junit.environment;
017
018import static lombok.AccessLevel.PRIVATE;
019
020import java.io.File;
021import java.net.MalformedURLException;
022import java.net.URL;
023import java.util.concurrent.ConcurrentHashMap;
024import java.util.concurrent.ConcurrentMap;
025import java.util.stream.Stream;
026
027import org.jboss.shrinkwrap.resolver.api.maven.ConfigurableMavenResolverSystem;
028import org.jboss.shrinkwrap.resolver.api.maven.Maven;
029import org.jboss.shrinkwrap.resolver.api.maven.MavenResolveStageBase;
030import org.jboss.shrinkwrap.resolver.api.maven.ScopeType;
031import org.jboss.shrinkwrap.resolver.api.maven.strategy.AcceptScopesStrategy;
032import org.jboss.shrinkwrap.resolver.api.maven.strategy.MavenResolutionStrategy;
033
034import lombok.NoArgsConstructor;
035
036@NoArgsConstructor(access = PRIVATE)
037public class Dependencies {
038
039    private static final ConcurrentMap<String, URL[]> CACHE = new ConcurrentHashMap<>();
040
041    private static final MavenResolveStageBase<?, ?, ?> RESOLVER = createResolver();
042
043    private static MavenResolveStageBase<?, ?, ?> createResolver() {
044        ConfigurableMavenResolverSystem resolverSystem =
045                Maven.configureResolver().withClassPathResolution(true).workOffline(
046                        Boolean.getBoolean("talend.component.junit.maven.offline"));
047
048        final String repos = System.getProperty("talend.component.junit.maven.repositories");
049        if (repos != null && !repos.isEmpty()) {
050            for (final String repo : repos.split(",")) {
051                final String[] parts = repo.split("::");
052                resolverSystem = resolverSystem.withRemoteRepo(parts[0], parts[1], parts[2]);
053            }
054        } else if (!Boolean.getBoolean("talend.component.junit.maven.skipPomReading")) {
055            // try to grab from the root pom the repos - simplifying maven parsing for speed reasons and enough
056            // generally
057
058            File current = new File(".").getAbsoluteFile();
059            while (new File(current, "pom.xml").exists()) {
060                final File parent = current.getParentFile();
061                if (parent != null && new File(parent, "pom.xml").exists()) {
062                    current = parent;
063                    continue;
064                }
065                break;
066            }
067            final File pom = new File(current, "pom.xml");
068            if (pom.exists()) {
069                return resolverSystem.loadPomFromFile(pom);
070            }
071        }
072
073        return resolverSystem;
074    }
075
076    private static final MavenResolutionStrategy STRATEGY =
077            new AcceptScopesStrategy(ScopeType.COMPILE, ScopeType.RUNTIME);
078
079    public static URL[] resolve(final String dep) {
080        return CACHE.computeIfAbsent(dep,
081                d -> Stream.of(RESOLVER.resolve(d).using(STRATEGY).asFile()).distinct().map(f -> {
082                    try {
083                        return f.toURI().toURL();
084                    } catch (final MalformedURLException e) {
085                        throw new IllegalStateException(e);
086                    }
087                }).toArray(URL[]::new));
088    }
089}
