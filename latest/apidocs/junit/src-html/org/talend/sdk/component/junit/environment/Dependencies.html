016package org.talend.sdk.component.junit.environment;
017
018import static lombok.AccessLevel.PRIVATE;
019
020import java.net.MalformedURLException;
021import java.net.URL;
022import java.util.concurrent.ConcurrentHashMap;
023import java.util.concurrent.ConcurrentMap;
024import java.util.stream.Stream;
025
026import org.jboss.shrinkwrap.resolver.api.maven.ConfigurableMavenResolverSystem;
027import org.jboss.shrinkwrap.resolver.api.maven.Maven;
028import org.jboss.shrinkwrap.resolver.api.maven.ScopeType;
029import org.jboss.shrinkwrap.resolver.api.maven.strategy.AcceptScopesStrategy;
030import org.jboss.shrinkwrap.resolver.api.maven.strategy.MavenResolutionStrategy;
031
032import lombok.NoArgsConstructor;
033
034@NoArgsConstructor(access = PRIVATE)
035public class Dependencies {
036
037    private static final ConcurrentMap<String, URL[]> CACHE = new ConcurrentHashMap<>();
038
039    private static final ConfigurableMavenResolverSystem RESOLVER = Maven.configureResolver();
040
041    private static final MavenResolutionStrategy STRATEGY =
042            new AcceptScopesStrategy(ScopeType.COMPILE, ScopeType.RUNTIME);
043
044    public static URL[] resolve(final String dep) {
045        return CACHE.computeIfAbsent(dep,
046                d -> Stream.of(RESOLVER.resolve(d).using(STRATEGY).asFile()).distinct().map(f -> {
047                    try {
048                        return f.toURI().toURL();
049                    } catch (final MalformedURLException e) {
050                        throw new IllegalStateException(e);
051                    }
052                }).toArray(URL[]::new));
053    }
054}
