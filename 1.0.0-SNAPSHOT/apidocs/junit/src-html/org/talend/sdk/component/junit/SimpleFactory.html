043@NoArgsConstructor(access = PRIVATE)
044public class SimpleFactory {
045
046    public static <T> Map<String, String> configurationByExample(final T instance) {
047        return configurationByExample(instance, "configuration.");
048    }
049
050    public static <T> Map<String, String> configurationByExample(final T instance, final String prefix) {
051        if (instance == null) {
052            return emptyMap();
053        }
054        final ParameterMeta params = new SimpleParameterModelService().build(prefix, prefix, instance.getClass(),
055                new Annotation[0], new ArrayList<>(singletonList(instance.getClass().getPackage().getName())));
056        return computeConfiguration(params.getNestedParameters(), instance, new HashMap<>());
057    }
058
059    private static Map<String, String> computeConfiguration(final List<ParameterMeta> nestedParameters,
060            final Object instance, final Map<Integer, Integer> indexes) {
061        if (nestedParameters == null) {
062            return emptyMap();
063        }
064        return nestedParameters.stream().map(param -> {
065            final Object value = getValue(instance, param.getName());
066            if (value == null) {
067                return Collections.<String, String> emptyMap();
068            }
069
070            switch (param.getType()) {
071            case OBJECT:
072                return computeConfiguration(param.getNestedParameters(), value, indexes);
073            case ARRAY:
074                final Collection<Object> values = Collection.class.isInstance(value) ? Collection.class.cast(value)
075                        : /* array */asList(Object[].class.cast(value));
076                final int arrayIndex = indexes.keySet().size();
077                final AtomicInteger valuesIndex = new AtomicInteger(0);
078                final Map<String, String> config = values.stream().map((Object item) -> {
079                    indexes.put(arrayIndex, valuesIndex.getAndIncrement());
080                    final Map<String, String> res =
081                            param.getNestedParameters().stream().filter(SimpleFactory::isPrimitive).collect(
082                                    toMap(p -> evaluateIndexes(p.getPath(), indexes),
083                                            p -> getValue(item, p.getName()).toString()));
084
085                    res.putAll(computeConfiguration(
086                            param.getNestedParameters().stream().filter(p -> !isPrimitive(p)).collect(toList()), item,
087                            indexes));
088                    return res;
089                }).flatMap(m -> m.entrySet().stream()).collect(toMap(Map.Entry::getKey, Map.Entry::getValue));
090
091                indexes.clear();// clear index after the end of array handling
092                return config;
093
094            default: // primitives
095                return singletonMap(evaluateIndexes(param.getPath(), indexes), value.toString());
096            }
097        }).flatMap(m -> m.entrySet().stream()).collect(toMap(Map.Entry::getKey, Map.Entry::getValue));
098    }
099
100    private static boolean isPrimitive(final ParameterMeta next) {
101        return Stream
102                .of(ParameterMeta.Type.STRING, ParameterMeta.Type.BOOLEAN, ParameterMeta.Type.ENUM,
103                        ParameterMeta.Type.NUMBER)
104                .anyMatch(v -> v == next.getType());
105    }
106
107    private static Object getValue(final Object instance, final String name) {
108        if (name.endsWith("[${index}]")) {
109            return instance;
110        }
111
112        Field declaredField = null;
113        Class<?> current = instance.getClass();
114        while (current != null && current != Object.class) {
115            try {
116                declaredField = current.getDeclaredField(name);
117                break;
118            } catch (final NoSuchFieldException e) {
119                current = current.getSuperclass();
120            }
121        }
122        if (declaredField == null) {
123            throw new IllegalArgumentException("No field '" + name + "' in " + instance);
124        }
125        if (!declaredField.isAccessible()) {
126            declaredField.setAccessible(true);
127        }
128        try {
129            return declaredField.get(instance);
130        } catch (final IllegalAccessException e) {
131            throw new IllegalStateException(e);
132        }
133    }
134
135    private static String evaluateIndexes(final String path, final Map<Integer, Integer> indexes) {
136        if (indexes == null || indexes.isEmpty()) {
137            return path;
138        }
139        String placeholder = "${index}";
140        String p = path;
141        StringBuilder evaluatedPath = new StringBuilder();
142        for (Map.Entry<Integer, Integer> index : indexes.entrySet()) {
143            int i = p.indexOf(placeholder);
144            evaluatedPath.append(p.substring(0, i)).append(index.getValue()).append("]");
145            p = p.substring(i + placeholder.length() + 1);
146        }
147        return evaluatedPath.append(p).toString();
148    }
149
150    private static class SimpleParameterModelService extends ParameterModelService {
151
152        private ParameterMeta build(final String name, final String prefix, final Type genericType,
153                final Annotation[] annotations, final Collection<String> i18nPackages) {
154            return super.buildParameter(name, prefix, null, genericType, annotations, i18nPackages);
155        }
156    }
